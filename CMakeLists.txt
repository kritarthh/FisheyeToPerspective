cmake_minimum_required(VERSION 2.8)
project( test )
find_package( OpenCV REQUIRED )
find_package(OpenMP)
if (OPENMP_FOUND)
    set (CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
    set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
endif()
add_executable( fishToPersp fishToPersp.cpp)
add_executable( cropAndResize cropAndResize.cpp)
target_link_libraries( fishToPersp ${OpenCV_LIBS} )
target_link_libraries( cropAndResize ${OpenCV_LIBS} )
